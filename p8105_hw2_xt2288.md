``` r
snp_df = 
  read_csv("data/snp.csv") |>
  janitor::clean_names() |>
  separate(date, into = c("month", "day","year")) |>
  mutate(
    year = as.numeric(year),
    year = ifelse( year <= 15, year + 2000, year + 1900),
    month = as.numeric(month)
  ) |>
  arrange(year, month) |>
```
